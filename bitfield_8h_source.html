105 const bitfield_c<bits> & operator=(const char * val) {
106
107 size_t l = strlen(val);
108
109 bt_assert(4*l <= bits);
110
111 int fpos = 0;
112 int bitpos = 0;
113 field[0] = 0;
114
115 for (int i = l-1; i >= 0; i--) {
116 uint64_t cval = 0;
117
118 if ((val[i] >= '0') && (val[i] <= '9'))
119 cval = val[i]-'0';
120 else if ((val[i] >= 'a') && (val[i] <= 'f'))
121 cval = val[i] - 'a' + 10;
122 else if ((val[i] >= 'A') && (val[i] <= 'F'))
123 cval = val[i] - 'A' + 10;
124 else
125 bt_assert(0);
126
127 field[fpos] |= cval << bitpos;
128
129 bitpos += 4;
130 if (bitpos >= 64) {
131 bitpos = 0;
132 fpos++;
133 field[fpos] = 0;
134 }
135 }
