41class gzstreambuf : public std::streambuf {
42private:
43 static const int bufferSize = 47+256; // size of data buff
44 // totals 512 bytes under g++ for igzstream at the end.
45
46 gzFile file; // file handle for compressed file
47 char buffer[bufferSize]; // data buffer
48 char opened; // open/close state of stream
49 int mode; // I/O mode
50
51 int flush_buffer();
52public:
53 gzstreambuf() : opened(0) {
54 setp( buffer, buffer + (bufferSize-1));
55 setg( buffer + 4, // beginning of putback area
56 buffer + 4, // read position
57 buffer + 4); // end position
58 // ASSERT: both input & output capabilities will not be used together
59 }
60 int is_open() { return opened; }
61 gzstreambuf* open( const char* name, int open_mode);
62 gzstreambuf* close();
63 ~gzstreambuf() { close(); }
64
65 virtual int overflow( int c = EOF);
66 virtual int underflow();
67 virtual int sync();
68};

83// User classes. Use igzstream and ogzstream analogously to ifstream and

88class igzstream : public gzstreambase, public std::istream {
