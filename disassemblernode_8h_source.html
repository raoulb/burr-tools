278 dat[4*i+0] = comefrom->dat[4*i+0]+tx;
279 dat[4*i+1] = comefrom->dat[4*i+1]+ty;
280 dat[4*i+2] = comefrom->dat[4*i+2]+tz;
281 dat[4*i+3] = comefrom->dat[4*i+3];
282 hashValue = 0;
